CREATE TABLE `reports` (
 `ReportID` int(11) NOT NULL AUTO_INCREMENT,
 `Status` enum('Active','Closed','On Hold') NOT NULL DEFAULT 'On Hold',
 `SubmissionDate` date NOT NULL,
 `FirstName` varchar(256) NOT NULL,
 `LastName` varchar(256) NOT NULL,
 `Gender` enum('Male','Female') NOT NULL,
 `Ethnicity` enum('American Indian','Asian','African American','Hispanic/Latino','Pacific Islander','White') NOT NULL,
 `EyeColor` enum('Brown','Blue','Amber','Gray','Green','Hazel','Red','Black') NOT NULL,
 `HairColor` varchar(256) NOT NULL,
 `Marks` varchar(256) DEFAULT NULL,
 `Weight` int(11) NOT NULL,
 `Height` int(11) NOT NULL,
 `DoB` date NOT NULL,
 `Phone` int(10) DEFAULT NULL,
 `SocialMedia` varchar(256) DEFAULT NULL,
 `Misc` text,
 `UserID` int(11) NOT NULL,
 `PlacesID1` int(11) DEFAULT NULL,
 `PlacesID2` int(11) DEFAULT NULL,
 `PlacesID3` int(11) DEFAULT NULL,
 `FriendsID1` int(11) DEFAULT NULL,
 `FriendsID2` int(11) DEFAULT NULL,
 `FriendsID3` int(11) DEFAULT NULL,
 `CaseNumber` int(11) NOT NULL,
 PRIMARY KEY (`ReportID`),
 KEY `UserID` (`UserID`),
 KEY `PlacesID1` (`PlacesID1`),
 KEY `PlacesID2` (`PlacesID2`),
 KEY `PlacesID3` (`PlacesID3`),
 KEY `FriendsID1` (`FriendsID1`),
 KEY `FriendsID2` (`FriendsID2`),
 KEY `FriendsID3` (`FriendsID3`),
 CONSTRAINT `Persons1` FOREIGN KEY (`FriendsID1`) REFERENCES `persons` (`PersonsID`) ON DELETE CASCADE ON UPDATE CASCADE,
 CONSTRAINT `Persons2` FOREIGN KEY (`FriendsID2`) REFERENCES `persons` (`PersonsID`) ON DELETE CASCADE ON UPDATE CASCADE,
 CONSTRAINT `Persons3` FOREIGN KEY (`FriendsID3`) REFERENCES `persons` (`PersonsID`) ON DELETE CASCADE ON UPDATE CASCADE,
 CONSTRAINT `Places1` FOREIGN KEY (`PlacesID1`) REFERENCES `places` (`PlacesID`) ON DELETE CASCADE ON UPDATE CASCADE,
 CONSTRAINT `Places2` FOREIGN KEY (`PlacesID2`) REFERENCES `places` (`PlacesID`) ON DELETE CASCADE ON UPDATE CASCADE,
 CONSTRAINT `Places3` FOREIGN KEY (`PlacesID3`) REFERENCES `places` (`PlacesID`) ON DELETE CASCADE ON UPDATE CASCADE,
 CONSTRAINT `User` FOREIGN KEY (`UserID`) REFERENCES `users` (`UserID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=latin1
